def count_consonants_in_rev(string):
  reversed_string = string[::-1]
  vowels = set('aeiou')
  count = 0
  for char in reversed_string:
    if string.isalpha() AND char in vowels:
      count += 1

  return count

count_consonants_in_rev("hello")

# The output should be 3
